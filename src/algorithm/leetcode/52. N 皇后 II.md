# 52. N 皇后 II

n 皇后问题 研究的是如何将 n 个皇后放置在 n × n 的棋盘上，并且使皇后彼此之间不能相互攻击。
给你一个整数 n ，返回 n 皇后问题 不同的解决方案的数量。

示例 1：
输入：n = 4
输出：2
解释：如上图所示，4 皇后问题存在两个不同的解法。

示例 2：
输入：n = 1
输出：1

```cpp
class Solution {
public:
    int totalNQueens(int n) {
        vector<bool> colVisited(n, false), diag1Visited(2 * n, false), diag2Visited(2 * n, false);
        int result = 0;
        search(n, 0, colVisited, diag1Visited, diag2Visited, result);
        return result;
    }

private:
    void search(int n, int rowIndex, vector<bool>& colVisited,
        vector<bool>& diag1Visited,
        vector<bool>& diag2Visited, int& result) {
        if (rowIndex == n) {
            ++result;
            return;
        }
        for (int i = 0; i < n; ++i) {
            if (colVisited[i] == false && diag1Visited[rowIndex + i] == false
                && diag2Visited[rowIndex - i + n] == false) {
                colVisited[i] = true;
                diag1Visited[rowIndex + i] = true;
                diag2Visited[rowIndex - i + n] = true;
                search(n, rowIndex + 1, colVisited, diag1Visited, diag2Visited, result);
                colVisited[i] = false;
                diag1Visited[rowIndex + i] = false;
                diag2Visited[rowIndex - i + n] = false;
            }
        }
    }
};
```
