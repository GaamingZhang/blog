# 15. 三数之和

```
给你一个整数数组 nums ，判断是否存在三元组 [nums[i], nums[j], nums[k]] 满足 i != j、i != k 且 j != k ，同时还满足 nums[i] + nums[j] + nums[k] == 0 。请你返回所有和为 0 且不重复的三元组。
注意：答案中不可以包含重复的三元组。

示例 1：
输入：nums = [-1,0,1,2,-1,-4]
输出：[[-1,-1,2],[-1,0,1]]
解释：
nums[0] + nums[1] + nums[2] = (-1) + 0 + 1 = 0 。
nums[1] + nums[2] + nums[4] = 0 + 1 + (-1) = 0 。
nums[0] + nums[3] + nums[4] = (-1) + 2 + (-1) = 0 。
不同的三元组是 [-1,0,1] 和 [-1,-1,2] 。
注意，输出的顺序和三元组的顺序并不重要。
```

```cpp
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        sort(nums.begin(), nums.end(), less<int>());
        vector<vector<int>> res;
        int index = 0;
        while (index < nums.size() - 2) {
            vector<vector<int>> tmpTwoSum = twoSum(nums, index + 1, nums.size() - 1, -nums[index]);
            for (int i = 0; i < tmpTwoSum.size(); ++i) {
                tmpTwoSum[i].push_back(nums[index]);
                res.push_back(tmpTwoSum[i]);
            }
            ++index;
            while (index < nums.size() && nums[index] == nums[index - 1]) {
                ++index;
            }
        }
        return res;
    }

    vector<vector<int>> twoSum(vector<int>& nums, int startIndex, int endIndex, int target) {
        vector<vector<int>> res;
        while (startIndex < endIndex) {
            if (nums[startIndex] + nums[endIndex] == target) {
                res.push_back({nums[startIndex], nums[endIndex]});
                ++startIndex;
                while (startIndex < endIndex && nums[startIndex] == nums[startIndex - 1]) {
                    ++startIndex;
                }
                --endIndex;
                while (startIndex < endIndex && nums[endIndex] == nums[endIndex + 1]) {
                    --endIndex;
                }
            } else if (nums[startIndex] + nums[endIndex] < target) {
                ++startIndex;
            } else {
                --endIndex;
            }
        }
        return res;
    }
};
