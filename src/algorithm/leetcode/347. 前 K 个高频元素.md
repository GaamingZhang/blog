# 347. 前 K 个高频元素

给你一个整数数组 nums 和一个整数 k ，请你返回其中出现频率前 k 高的元素。你可以按 任意顺序 返回答案。

示例 1：
输入：nums = [1,1,1,2,2,3], k = 2
输出：[1,2]

示例 2：
输入：nums = [1], k = 1
输出：[1]

示例 3：
输入：nums = [1,2,1,2,1,2,3,1,3,2], k = 2
输出：[1,2]

```cpp
class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        unordered_map<int, int> occurrences;
        for (int i = 0; i < nums.size(); ++i) {
            ++occurrences[nums[i]];
        }
        vector<pair<int, int>> occurrencesArray;
        for (auto& [num, count] : occurrences) {
            occurrencesArray.push_back({num, count});
        }
        topKFrequent(occurrencesArray, 0, occurrencesArray.size() - 1, k);
        vector<int> result;
        for (int i = 0; i < k; ++i) {
            result.push_back(occurrencesArray[i].first);
        }
        return result;
    }

private:
    void topKFrequent(vector<pair<int, int>>& nums, int begin, int end, int k) {
        if (begin >= end)
            return;
        int randIndex = begin + rand() % (end - begin + 1);
        int privot = nums[randIndex].second, iterBegin = begin - 1,
            iterEnd = end + 1, iter = begin;
        while (iter < iterEnd) {
            if (nums[iter].second > privot) {
                swap(nums[++iterBegin], nums[iter++]);
            } else if (nums[iter].second < privot) {
                swap(nums[--iterEnd], nums[iter]);
            } else
                ++iter;
        }
        int pivotIndexBegin = iterBegin + 1, pivotIndexEnd = iterEnd - 1;
        if (pivotIndexBegin + 1 <= k && pivotIndexEnd + 1 >= k)
            return;
        else if (pivotIndexBegin + 1 > k)
            topKFrequent(nums, begin, pivotIndexBegin - 1, k);
        else
            topKFrequent(nums, pivotIndexEnd + 1, end, k);
    }
};
```
