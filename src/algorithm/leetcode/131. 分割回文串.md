# 131. 分割回文串

给你一个字符串 s，请你将 s 分割成一些 子串，使每个子串都是 回文串 。返回 s 所有可能的分割方案。

示例 1：
输入：s = "aab"
输出：[["a","a","b"],["aa","b"]]

示例 2：
输入：s = "a"
输出：[["a"]]

```cpp
class Solution {
public:
    vector<vector<string>> partition(string s) {
        vector<vector<string>> result;
        vector<string> tmpResult;
        string tmp;
        partition(s, result, tmpResult);
        return result;
    }

private:
    void partition(string s, vector<vector<string>>& result,
                   vector<string>& tmpResult) {
        if (s.empty()) {
            result.push_back(tmpResult);
            return;
        }
        for (int i = 1; i <= s.size(); ++i) {
            string tmpString = s.substr(0, i);
            if (isPartition(tmpString)) {
                tmpResult.push_back(tmpString);
                partition(s.substr(i, s.size() - i), result, tmpResult);
                tmpResult.pop_back();
            }
        }
    }
    bool isPartition(string s) {
        if (s.size() == 0)
            return false;
        int begin = 0, end = s.size() - 1;
        while (begin < end) {
            if (s[begin++] != s[end--])
                return false;
        }
        return true;
    }
};
```
