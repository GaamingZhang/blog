# 4. 寻找两个正序数组的中位数

给定两个大小分别为 m 和 n 的正序（从小到大）数组 nums1 和 nums2。请你找出并返回这两个正序数组的 中位数 。

算法的时间复杂度应该为 O(log (m+n)) 。

示例 1：
输入：nums1 = [1,3], nums2 = [2]
输出：2.00000
解释：合并数组 = [1,2,3] ，中位数 2

示例 2：
输入：nums1 = [1,2], nums2 = [3,4]
输出：2.50000
解释：合并数组 = [1,2,3,4] ，中位数 (2 + 3) / 2 = 2.5

```cpp
class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size(), m = nums2.size();
        if ((n + m) % 2) {
            return findKth(nums1, nums2, (n + m) / 2 + 1);
        } else {
            return (findKth(nums1, nums2, (n + m) / 2) +
                    findKth(nums1, nums2, (n + m) / 2 + 1)) /
                   2.0;
        }
    }

private:
    int findKth(vector<int>& nums1, vector<int>& nums2, int k) {
        int n = nums1.size(), m = nums2.size(), index1 = 0, index2 = 0;
        while (true) {
            if (index1 == n)
                return nums2[index2 + k - 1];
            if (index2 == m)
                return nums1[index1 + k - 1];
            if (k == 1)
                return min(nums1[index1], nums2[index2]);
            int nextIndex1 = min(index1 + k / 2 - 1, n - 1);
            int nextIndex2 = min(index2 + k / 2 - 1, m - 1);
            if (nums1[nextIndex1] < nums2[nextIndex2]) {
                k -= nextIndex1 - index1 + 1;
                index1 = nextIndex1 + 1;
            } else {
                k -= nextIndex2 - index2 + 1;
                index2 = nextIndex2 + 1;
            }
        }
    }
};
```
