# 17. 电话号码的字母组合

给定一个仅包含数字 2-9 的字符串，返回所有它能表示的字母组合。答案可以按 任意顺序 返回。
给出数字到字母的映射如下（与电话按键相同）。注意 1 不对应任何字母。

示例 1：
输入：digits = "23"
输出：["ad","ae","af","bd","be","bf","cd","ce","cf"]

示例 2：
输入：digits = ""
输出：[]

示例 3：
输入：digits = "2"
输出：["a","b","c"]

```cpp
class Solution {
public:
    vector<string> letterCombinations(string digits) {
        if (digits.size() == 0) {
            return result;
        }
        letterCombinations(digits, 0, "");
        return result;
    }

private:
    vector<vector<string>> numbers = {
        {""},
        {"a", "b", "c"},
        {"d", "e", "f"},
        {"g", "h", "i"},
        {"j", "k", "l"},
        {"m", "n", "o"},
        {"p", "q", "r", "s"},
        {"t", "u", "v"},
        {"w", "x", "y", "z"}
    };
    vector<string> result;

    void letterCombinations(string digits, int index, string tmpRes) {
        if (digits.size() == index) {
            result.push_back(tmpRes);
            return;
        }
        int wordIndex = digits[index] - '1';
        for (int i = 0; i < numbers[wordIndex].size(); ++i) {
            letterCombinations(digits, index + 1, tmpRes + numbers[wordIndex][i]);
        }
    }
};
```
