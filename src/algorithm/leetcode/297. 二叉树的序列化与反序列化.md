# 297. 二叉树的序列化与反序列化

序列化是将一个数据结构或者对象转换为连续的比特位的操作，进而可以将转换后的数据存储在一个文件或者内存中，同时也可以通过网络传输到另一个计算机环境，采取相反方式重构得到原数据。

请设计一个算法来实现二叉树的序列化与反序列化。这里不限定你的序列 / 反序列化算法执行逻辑，你只需要保证一个二叉树可以被序列化为一个字符串并且将这个字符串反序列化为原始的树结构。

提示: 输入输出格式与 LeetCode 目前使用的方式一致，详情请参阅 LeetCode 序列化二叉树的格式。你并非必须采取这种方式，你也可以采用其他的方法解决这个问题。

示例 1：
输入：root = [1,2,3,null,null,4,5]
输出：[1,2,3,null,null,4,5]

示例 2：
输入：root = []
输出：[]

示例 3：
输入：root = [1]
输出：[1]

示例 4：
输入：root = [1,2]
输出：[1,2]

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Codec {
public:
    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        queue<TreeNode*> q;
        q.push(root);
        string result;
        while (!q.empty()) {
            TreeNode* currentNode = q.front();
            q.pop();
            if (currentNode == nullptr) {
                result += "100010,";
                continue;
            }
            result += to_string(currentNode->val);
            result += ",";
            q.push(currentNode->left);
            q.push(currentNode->right);
        }
        return result;
    }

    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        vector<int> values;
        int index = 0, pre = 0;
        while (index < data.size()) {
            while (index < data.size() && data[index] != ',')
                ++index;
            values.push_back(stoi(data.substr(pre, index - pre)));
            pre = ++index;
        }
        if (values.size() == 0)
            return nullptr;
        if (values[0] == 100010)
            return nullptr;

        queue<TreeNode*> q;
        index = 0;
        TreeNode* head = new TreeNode(values[index++]);
        q.push(head);
        while (!q.empty()) {
            TreeNode* currentNode = q.front();
            q.pop();
            if (values[index] != 100010) {
                currentNode->left = new TreeNode(values[index]);
                q.push(currentNode->left);
            }
            ++index;
            if (values[index] != 100010) {
                currentNode->right = new TreeNode(values[index]);
                q.push(currentNode->right);
            }
            ++index;
        }
        return head;
    }
};
```
