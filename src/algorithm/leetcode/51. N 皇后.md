# 51. N 皇后
按照国际象棋的规则，皇后可以攻击与之处在同一行或同一列或同一斜线上的棋子。

n 皇后问题 研究的是如何将 n 个皇后放置在 n×n 的棋盘上，并且使皇后彼此之间不能相互攻击。

给你一个整数 n ，返回所有不同的 n 皇后问题 的解决方案。

每一种解法包含一个不同的 n 皇后问题 的棋子放置方案，该方案中 'Q' 和 '.' 分别代表了皇后和空位。

示例 1：
输入：n = 4
输出：[[".Q..","...Q","Q...","..Q."],["..Q.","Q...","...Q",".Q.."]]
解释：如上图所示，4 皇后问题存在两个不同的解法。

示例 2：
输入：n = 1
输出：[["Q"]]

```cpp
class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        vector<vector<bool>> board(n, vector<bool>(n, false));
        vector<vector<string>> results;
        vector<int> visitedRow(n, 0), visitedCol(n, 0), diagonals1(2 * n, 0),
            diagonals2(2 * n, 0);
        search(board, results, n, 0, visitedRow, visitedCol, diagonals1,
               diagonals2);
        return results;
    }

private:
    bool check(int& x, int& y, int& n, vector<int>& visitedRow,
               vector<int>& visitedCol, vector<int>& diagonals1,
               vector<int>& diagonals2) {
        if (visitedRow[x] > 0 || visitedCol[y] > 0 || diagonals1[x + y] > 0 ||
            diagonals2[x - y + n] > 0)
            return false;
        return true;
    }
    vector<string> boardToResult(vector<vector<bool>>& board, int& n) {
        vector<string> result;
        for (int i = 0; i < n; ++i) {
            string row;
            for (int j = 0; j < n; ++j) {
                if (board[i][j])
                    row += "Q";
                else
                    row += ".";
            }
            result.push_back(row);
        }
        return result;
    }
    void search(vector<vector<bool>>& board, vector<vector<string>>& results,
                int n, int rowIndex, vector<int>& visitedRow,
                vector<int>& visitedCol, vector<int>& diagonals1,
                vector<int>& diagonals2) {
        if (rowIndex == n) {
            vector<string> result = boardToResult(board, n);
            results.push_back(result);
            return;
        }
        for (int colIndex = 0; colIndex < n; ++colIndex) {
            board[rowIndex][colIndex] = true;
            if (check(rowIndex, colIndex, n, visitedRow, visitedCol, diagonals1,
                      diagonals2)) {
                visitedRow[rowIndex] += 1;
                visitedCol[colIndex] += 1;
                diagonals1[rowIndex + colIndex] += 1;
                diagonals2[rowIndex - colIndex + n] += 1;
                search(board, results, n, rowIndex + 1, visitedRow, visitedCol,
                       diagonals1, diagonals2);
                visitedRow[rowIndex] -= 1;
                visitedCol[colIndex] -= 1;
                diagonals1[rowIndex + colIndex] -= 1;
                diagonals2[rowIndex - colIndex + n] -= 1;
            }
            board[rowIndex][colIndex] = false;
        }
    }
};
```
