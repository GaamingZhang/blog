# 68. 文本左右对齐

给定一个单词数组 words 和一个长度 maxWidth ，重新排版单词，使其成为每行恰好有 maxWidth 个字符，
 且左右两端对齐的文本。
你应该使用 "贪心算法" 来放置给定的单词；也就是说，尽可能多地往每行中放置单词。必要时可用空格 ' '
 填充，使得每行恰好有 maxWidth 个字符。
要求尽可能均匀分配单词间的空格数量。如果某一行单词间的空格不能均匀分配，则左侧放置的空格数
 要多于右侧的空格数。
文本的最后一行应为左对齐，且单词之间不插入额外的空格。

```cpp
class Solution {
public:
    vector<string> fullJustify(vector<string>& words, int maxWidth) {
        vector<string> result;
        int i = 0;
        vector<string> row;
        int rowSize = 0;
        while (i < words.size()) {
            int currentSize = rowSize + words[i].size() + row.size();
            if (currentSize > maxWidth) {
                result.push_back(getRow(row, rowSize, maxWidth, false));
                row.clear();
                rowSize = 0;
            }
            row.push_back(words[i]);
            rowSize += words[i].size();
            ++i;
        }
        result.push_back(getRow(row, rowSize, maxWidth, true));
        return result;
    }

private:
    string getRow(vector<string>& row, int rowSize, int maxWidth, bool lastLine) {
        string result;
        if (lastLine) {
            for (int i = 0; i < row.size(); ++i) {
                result += row[i];
                maxWidth -= row[i].size();
                if (i != row.size() - 1) {
                    result += " ";
                    --maxWidth;
                }
            }
            while (maxWidth-- > 0) {
                result += " ";
            }
            return result;
        }
        int leftSize = maxWidth - rowSize;
        int eachSpace = row.size() == 1 ? 0 : leftSize / (row.size() - 1);
        int leftSpace = row.size() == 1 ? leftSize : leftSize % (row.size() - 1);
        for (int i = 0; i < row.size(); ++i) {
            result += row[i];
            if (i != row.size() - 1) {
                for (int j = 0; j < eachSpace; ++j) {
                    result += " ";
                }
            }
            if (leftSpace > 0) {
                result += " ";
            }
            --leftSpace;
        }
        while (leftSpace-- > 0) {
            result += " ";
        }
        return result;
    }
};
```
